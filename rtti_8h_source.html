153  void *DirectCast(void *src_obj, TypeId src, TypeId dst)
154  {
155  if (src == dst)
156  return src_obj;
157  auto src_iter = m_casts.find(src);
158  if (src_iter != m_casts.end())
159  {
160  auto &&dst_to_fn = src_iter->second;
161  auto dst_iter = dst_to_fn.find(dst);
162  if (dst_iter != dst_to_fn.end())
163  {
164  auto cast_fn = dst_iter->second;
165  return cast_fn(src_obj);
166  }
167  }
168  return nullptr;
169  }
170 
171  void *RecursiveCast(void *src_obj, TypeId src, TypeId dst, std::set<TypeId> &blacklist)
172  {
173  auto src_iter = m_casts.find(src);
174  if (src_iter != m_casts.end())
175  {
176  auto &&dst_to_fn_map = src_iter->second;
177  // for bridge in src.children:
178  // try cast: src -> bridge -> dst
179  for (auto &&pair : dst_to_fn_map)
180  {
181  const TypeId bridge = pair.first;
182 
183  if (blacklist.count(bridge))
184  continue;
185 
186  auto src_to_bridge_fn = pair.second;
187  void *bridge_obj = src_to_bridge_fn(src_obj);
188  if (bridge_obj == nullptr)
189  {
190  blacklist.insert(bridge);
191  continue;
192  }
193 
194  void *dst_obj = RecursiveCast(bridge_obj, bridge, dst, blacklist);
195  if (dst_obj)
196  {
197  return dst_obj;
198  }
199  else
200  {
201  blacklist.insert(bridge);
202  continue;
203  }
204  }
205  }
206  return nullptr;
207  }

262 inline constexpr bool LLAMA_RTTI_ArgCheck(const char *arg)
263 {
